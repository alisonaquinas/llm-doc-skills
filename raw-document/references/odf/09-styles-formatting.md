## 1. ODF Style Architecture

ODF uses a two-tier style system:

### Named Styles (User-visible)

Stored in `styles.xml`. These are the styles a user sees in the application (e.g., "Heading 1", "Default Paragraph Style"). Named styles can inherit from other styles.

### Automatic Styles (Application-generated)

Stored in `content.xml`. These are transient, unnamed styles generated by the application when direct formatting is applied. They are not visible to the user as named styles.

**Key principle**: Document interoperability is best achieved by using named styles rather than relying on automatic styles.

## 3. Style Element Structure

```xml
<style:style
    style:name="Heading_20_1"        <!-- internal style name (spaces as _20_) -->
    style:display-name="Heading 1"   <!-- user-visible name -->
    style:family="paragraph"
    style:parent-style-name="Heading" <!-- inherits from "Heading" -->
    style:list-style-name="Heading_List_Style"  <!-- associated list style -->
    style:next-style-name="Text_20_Body"        <!-- next paragraph style -->
    style:class="text">              <!-- optional style class -->

  <!-- Optional: list-level-style-image for list bullets -->

  <style:paragraph-properties
      fo:margin-top="0.4cm"
      fo:margin-bottom="0.2cm"
      fo:keep-with-next="always"
      fo:break-before="auto"
      fo:text-align="left"/>

  <style:text-properties
      fo:font-size="18pt"
      fo:font-weight="bold"
      fo:color="#2E4E8F"
      style:font-name="Calibri"
      fo:font-family="Calibri, sans-serif"/>

</style:style>
```

## 4. Paragraph Properties (`style:paragraph-properties`)

| Attribute | Description | Example Values |
| --- | --- | --- |
| `fo:text-align` | Horizontal alignment | `left`, `right`, `center`, `justify`, `start`, `end` |
| `fo:text-indent` | First-line indent | `1.25cm` |
| `fo:margin-left` | Left indent | `2cm` |
| `fo:margin-right` | Right indent | `0cm` |
| `fo:margin-top` | Space before paragraph | `0.2cm` |
| `fo:margin-bottom` | Space after paragraph | `0.2cm` |

| `fo:keep-together` | Prevent splitting | `auto`, `always` |
| `fo:keep-with-next` | Keep with next para | `auto`, `always` |
| `fo:break-before` | Page/column break before | `auto`, `page`, `column`, `even-page`, `odd-page` |
| `fo:break-after` | Page/column break after | `auto`, `page`, `column` |
| `fo:widows` | Minimum widow lines | `2` |
| `fo:orphans` | Minimum orphan lines | `2` |

| `fo:border` | Border shorthand | `0.05cm solid #000000` |

| `fo:padding` | Padding shorthand | `0.1cm` |
| `style:writing-mode` | Text direction | `lr-tb`, `rl-tb`, `tb-rl` |
| `style:snap-to-layout-grid` | Grid snapping | `true`, `false` |
| `text:number-lines` | Line numbering | `true`, `false` |

## 6. Page Layout and Master Pages

### Page Layout Definition

```xml
<!-- In styles.xml -->
<style:page-layout style:name="Mpm1">
  <style:page-layout-properties
      fo:page-width="21cm"
      fo:page-height="29.7cm"           <!-- A4 portrait -->
      style:print-orientation="portrait"
      fo:margin-top="2cm"
      fo:margin-bottom="2cm"
      fo:margin-left="2.5cm"
      fo:margin-right="2cm"
      fo:background-color="transparent"
      style:writing-mode="lr-tb"
      style:footnote-max-height="0cm">

    <!-- Footnote separator line -->
    <style:footnote-sep style:adjustment="left"
                        style:rel-width="25%"
                        style:line-style="solid"
                        style:width="0.018cm"/>

  </style:page-layout-properties>

  <style:header-style>
    <style:header-footer-properties
        fo:min-height="0.8cm"
        fo:margin-left="0cm"
        fo:margin-right="0cm"
        fo:margin-bottom="0.3cm"
        style:dynamic-spacing="false"/>
  </style:header-style>

  <style:footer-style>
    <style:header-footer-properties
        fo:min-height="0.6cm"
        fo:margin-left="0cm"
        fo:margin-right="0cm"
        fo:margin-top="0.3cm"/>
  </style:footer-style>
</style:page-layout>
```

### Master Page Definition

```xml
<style:master-page style:name="Default"
                   style:page-layout-name="Mpm1">

  <style:header>
    <text:p text:style-name="Header">
      <text:chapter text:display="name" text:outline-level="1"/>
    </text:p>
  </style:header>

  <style:header-left>
    <text:p text:style-name="Header">
      <text:file-name text:display="name"/>
    </text:p>
  </style:header-left>

  <style:footer>
    <text:p text:style-name="Footer">
      Page <text:page-number text:select-page="current">1</text:page-number>
      of <text:page-count>1</text:page-count>
    </text:p>
  </style:footer>

</style:master-page>
```

## 7. List Styles

```xml
<text:list-style style:name="List_20_Bullet">

  <!-- Level 1: bullet -->
  <text:list-level-style-bullet
      text:level="1"
      text:style-name="Bullet_20_Symbols"
      text:bullet-char="ŌĆó">
    <style:list-level-properties
        text:space-before="0.5cm"
        text:min-label-width="0.5cm"/>
  </text:list-level-style-bullet>

  <!-- Level 2: different bullet -->
  <text:list-level-style-bullet
      text:level="2"
      text:bullet-char="ŌĆō">
    <style:list-level-properties
        text:space-before="1cm"
        text:min-label-width="0.5cm"/>
  </text:list-level-style-bullet>

</text:list-style>


<text:list-style style:name="List_20_Number">

  <!-- Level 1: numbered -->
  <text:list-level-style-number
      text:level="1"
      text:style-name="Numbering_20_Symbols"
      style:num-format="1"            <!-- 1=decimal, a=lower-alpha, A=upper-alpha, i=lower-roman, I=upper-roman -->
      text:start-value="1">
    <style:list-level-properties
        text:space-before="0.5cm"
        text:min-label-width="0.5cm"/>
  </text:list-level-style-number>

</text:list-style>
```

## 8. Number (Data) Styles

Data styles control how numbers, dates, times, and currencies are displayed in cells:

```xml
<number:number-style style:name="N2">
  <number:number number:decimal-places="2" number:min-integer-digits="1"/>
</number:number-style>

<number:currency-style style:name="N4">
  <number:currency-symbol number:language="en" number:country="US">$</number:currency-symbol>
  <number:number number:decimal-places="2" number:grouping="true" number:min-integer-digits="1"/>
</number:currency-style>

<number:date-style style:name="N37">
  <number:month number:style="long"/>
  <number:text>/</number:text>
  <number:day number:style="long"/>
  <number:text>/</number:text>
  <number:year number:style="long"/>
</number:date-style>

<number:time-style style:name="N40">
  <number:hours number:style="long"/>
  <number:text>:</number:text>
  <number:minutes number:style="long"/>
  <number:text>:</number:text>
  <number:seconds number:style="long"/>
</number:time-style>
```

## 9. Default Styles

Default styles set the baseline formatting for each family:

```xml
<style:default-style style:family="paragraph">
  <style:paragraph-properties fo:margin-top="0cm" fo:margin-bottom="0cm"
                               fo:line-height="115%"
                               style:writing-mode="lr-tb"/>
  <style:text-properties style:font-name="Calibri"
                          fo:font-family="Calibri, sans-serif"
                          fo:font-size="11pt"
                          fo:language="en"
                          fo:country="US"
                          fo:color="#000000"/>
</style:default-style>

<style:default-style style:family="table-cell">
  <style:text-properties style:font-name="Calibri"
                          fo:font-size="11pt"/>
</style:default-style>
```

## 10. Style Inheritance

ODF styles support hierarchical inheritance through `style:parent-style-name`:

```text
Default Paragraph Style
  ŌööŌöĆŌöĆ Text Body
        ŌööŌöĆŌöĆ Text Body Indent
  ŌööŌöĆŌöĆ Heading
        ŌööŌöĆŌöĆ Heading 1
        ŌööŌöĆŌöĆ Heading 2
              ŌööŌöĆŌöĆ Heading 3
```

A child style inherits all properties from its parent, then overrides specific properties. This mirrors CSS inheritance.
